\begin{frame}{Εξίσωση 1}
 \begin{block}{$ημx=ημθ$}
  Η εξίσωση $ημx=ημθ$ έχει λύσεις
  $$x=2κπ+θ$$
  και
  $$x=2κπ+π-θ$$
  όπου $κ\in\mathbb{Z}$
 \end{block}
\end{frame}
