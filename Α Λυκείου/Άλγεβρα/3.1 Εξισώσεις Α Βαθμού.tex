\exercises

\subsection{Μορφή $αx+b=0$}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $x-2(x-1)=5-(1-3x)$
    \item $\dfrac{2x-1}{2}-\dfrac{4x-3}{4}=2$
    \item $\dfrac{3x-6}{3}= \dfrac{2x-4}{2}$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $(x-1)^2=1-x(3-2x)$
    \item $x^3-2x^2=9x-18$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $2-\dfrac{x^2+7x}{x^2-1}= \dfrac{2x-1}{x+1}+\dfrac{3}{1-x}$
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $\dfrac{x^2-1}{x^2-2x+1}-\dfrac{2x}{x^2-x}-1=0$
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $(λ-1)x=λ+1$, για τις διάφορες τιμές του $λ\in\mathbb{R}$.
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $λ(λ-x)-λx(λ-1)=2(λ-2x)$, για τις διάφορες τιμές του $λ\in\mathbb{R}$.
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $x^3+(x-1)^3-(2x-1)^3=0$
\end{askisi}

\begin{askisi}
  Να βρείτε τις τιμές του $λ$, για τις οποίες η εξίσωση
  $$λ(x-λ)=2(x-2)$$
  \begin{enumerate}[<+->]
    \item έχει λύση το $2$
    \item έχει μοναδική λύση το $2$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να βρείτε τις τιμές των $λ$, $μ$, για τις οποίες η εξίσωση
  $$λ(x-2)-1=x-μ$$
  \begin{enumerate}[<+->]
    \item είναι ταυτότητα
    \item είναι αδύνατη
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να δείξετε ότι για κάθε $α$, $β\in\mathbb{R}^*$, η παρακάτω εξίσωση έχει λύση
  $$\frac{x+α}{β}=\frac{x+β}{α}$$
\end{askisi}

\subsection{Εξίσώσεις με Απόλυτα}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $|x|-3=0$
    \item $1-2|x|=0$
    \item $3|x|+1=0$
    \item $d(2x,3)-1=0$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $|-x|+|2x|-6\left| \dfrac{x}{3} \right|=0$
    \item $\dfrac{|3x-2|}{2}-\dfrac{|2-3x|-3}{4}=3-|6x-4|$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $|3x-1|=|x-5|$
    \item $|2x-3|-2|x|=0$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $\sqrt{x^2-6x+9}=|x-1|$
    \item $\sqrt{x^2+1}=|x-1|$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $|x-1|=3x-7$
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $|x|-|x-1|+2x-3=0$
\end{askisi}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $|x-2|-x=-2$
    \item $d(x,3)=3-x$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $\left| |x-1| +1\right|=3$
    \item $\left| |x-2| -1\right|=1$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε τις εξισώσεις:
  \begin{enumerate}[<+->]
    \item $\dfrac{|x+1|}{|1-x|}=3$
    \item $|x-2|\cdot |x-3|=|x-2|$
    \item $x^2=3\sqrt{x^2}$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $|x^2-x|+|x^2-1|=0$
\end{askisi}

\begin{askisi}
  Να λύσετε την εξίσωση $|x|+1=\dfrac{1}{\sqrt{x^2+1}}$
\end{askisi}
