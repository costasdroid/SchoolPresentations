\section{Ασκήσεις}

\exercises

\begin{askisi}
  Αν $x_1, x_2$ είναι οι ρίζες της εξίσωσης $x^2 - 5x + 1 = 0$, να υπολογίσετε:
  \begin{enumerate}[<+->]
    \item $x_1\cdot x_2-x_1-x_2$
    \item $\dfrac{1}{x_1}+\dfrac{1}{x_2}$
    \item $x_1^2\cdot x_2+x_1\cdot x_2^2$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Αν $x_1, x_2$ είναι οι ρίζες της εξίσωσης $x^2 + 3x - 1=0$, να υπολογίσετε:
  \begin{enumerate}[<+->]
    \item $x_1^2 + x_2^2$
    \item $\dfrac{x_1}{x_2} + \dfrac{x_2}{x_1}$
    \item $(x_1-x_2)^2$
    \item $|x_1-x_2|$
    \item $x_1^3 + x_2^3$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να βρείτε την εξίσωση 2ου βαθμού που έχει ρίζες:
  \begin{enumerate}[<+->]
    \item $-1$ και $2$
    \item $-1$ και $\dfrac{2}{3}$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να βρείτε δύο αριθμούς, που έχουν:
  \begin{enumerate}[<+->]
    \item άθροισμα $-2$ και γινόμενο $-5$
    \item άθροισμα $5$ και γινόμενο $8$
  \end{enumerate}
\end{askisi}

\begin{askisi}
  Να προσδιόσετε τις ρίζες των παρακάτω εξισώσεων (χωρίς να τις λύσετε)
  \begin{enumerate}[<+->]
    \item $x^2 - 5x + 6 = 0$
    \item $x^2 + 4x -5 = 0$
  \end{enumerate}
\end{askisi}
